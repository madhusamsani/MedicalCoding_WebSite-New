const express = require('express'); const bodyParser = require('body-parser'); const nodemailer = require('nodemailer'); const app = express(); const PORT = process.env.PORT || 3000; app.use(bodyParser.urlencoded({ extended: true })); // Replace with your email provider config const transporter = nodemailer.createTransport({ service: 'gmail', auth: { user: 'madhu.samsani@gmail.com', // your Gmail address pass: 'Madhu@1234', // Gmail App Password, NOT your regular password }, }); app.post('/contact', (req, res) => { const { name, email, phone, msg } = req.body; const mailOptions = { from: `"${name}" <${email}>`, to: 'madhu.samsani@gmail.com', subject: 'New Contact Form Submission', text: `Name: ${name} Email: ${email} Phone: ${phone} Message: ${msg}`, }; transporter.sendMail(mailOptions, (error, info) => { if (error) { console.error(error); return res.status(500).send('Error sending email.'); } res.send('Thank you for contacting us!'); }); }); app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
